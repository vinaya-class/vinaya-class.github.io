\chapter{21.A. Sekhiyas 1}
\renewcommand*{\theChapterTitle}{21.A. Sekhiyas 1}

\begin{exam}{\autoExamName}

  \begin{problem*}
    \textbf{Do} or do \textbf{Not}?

    \bigskip

    \begin{parts}

    \item \TF{D} Walking on the road to the town wearing a sun hat on hot day.

    \bigskip

    \item \TF{N} Sitting in an angsa while travelling in a car.

    \bigskip

    \item \TF{D} Visiting the town hall, wearing the upper-robes on both shoulders.

    \bigskip

    \item \TF{N} Walking along a crowded beach in an angsa.

    \bigskip

    \item \TF{N} Sitting in an angsa in a public park.

    \bigskip

    \item \TF{N} Using a corn-field as cover for defecating.

    \bigskip

    \item \TF{N} Walking on the street, explaining a story and wildly gesticulating with the arms for emphasis.

    \bigskip

    \item \TF{D} Walking along a river, stopping to urinate, away from the river.

    \bigskip

    \item \TF{N} Being in a hurry before the \emph{uposatha-kamma}, pulling up the upper robe and urinating.

    \bigskip

    \item \TF{N/D} The air-conditioning inside a supermarket makes the air cold, so a bhikkhu puts on woolly hat.

    \begin{solution}
      In general it is not good \emph{samaṇasaññā}, so for a short period it is recommended to not wear hats in public spaces.

      If one has to stay under air-conditioning for a long period (e.g. flights, over-cooled waiting rooms), the cold can be damaging to one's health, and hats are acceptable.
    \end{solution}

    \bigskip

    \item \TF{N} Having parked and walked away from the car, yelling back to the driver to bring a water bottle.

    \bigskip

    \item \TF{D} A bhikkhu needs walking boots when travelling to the mountains.
    He packs the boots in the luggage and on the airport he is wearing sandals.

    \bigskip

    \item \TF{N} Diligently listening to Dhamma talks with head-phones while
    walking on alms-round.

    \bigskip

    \item \TF{N} Using head-phones efficiently and making calls while walking
    around the monastery on routine errands.

    \bigskip

    \item \TF{D} Teenagers on the street comment on a bhikkhu, `Jesus has come!'
    He continues walking in silence, with eyes lowered and composed in his
    posture.

    \end{parts}

  \end{problem*}

\end{exam}
